89 template<typename T>
90 TX_INLINE void jsonLoad(simdjson::ondemand::document& json_object, T& value) {
91  // iterate over the members of T
92  for_each(refl::reflect(value).members, [&](auto member) {
93  // is_readable checks if the member is a non-const field
94  // or a 0-arg const-qualified function marked with property attribute
95  if constexpr (!is_readable(member)) { return; }
96 
97  if constexpr (refl::descriptor::has_attribute<taixu::ISerializableStr>(
98  member)) {
99  member(value) = std::string(
100  json_object[get_display_name(member)].get_string().value());
101  } else if constexpr (refl::descriptor::has_attribute<
102  taixu::ISerializableNumber>(member)) {
103  member(value) = json_object[get_display_name(member)].get_double();
104  } else if constexpr (refl::descriptor::has_attribute<
105  taixu::ISerializableBool>(member)) {
106  member(value) = json_object[get_display_name(member)].get_bool();
107  } else if constexpr (refl::descriptor::has_attribute<
108  taixu::ISerializableObject>(member)) {
109  simdjson::ondemand::object data =
110  json_object[get_display_name(member)];
111  member(value) = jsonLoad(data, member(value));
112  } else if constexpr (refl::descriptor::has_attribute<
113  taixu::ISerializableList>(member)) {
114  simdjson::ondemand::array data =
115  json_object[get_display_name(member)];
116  std::size_t cnt = data.count_elements();
117  auto& member_ref = member(value);
118  member_ref.resize(cnt);
119  std::size_t idx = 0;
120  for (auto sub_data : data) {
121  member_ref[idx++] = jsonLoad(sub_data, member(value));
122  }
123  } else if constexpr (refl::descriptor::has_attribute<
124  taixu::ISerializableNULL>(member)) {
125  throw std::runtime_error("not support null");
126  } else {
127  throw std::runtime_error("not support type");
128  }
129  });
130 }

141  T value{};
