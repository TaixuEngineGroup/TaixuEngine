31  // Total size of valid entries in the array.

38  void insertData(const Key& key, const Value& value) noexcept {
39  if (contains(key)) {
40  _values[_key_to_index_map[key]] = value;
41  return;
42  }
43  std::size_t new_idx = _size;
44  _key_to_index_map[key] = new_idx;

48  }

62  void removeData(const Key& key) {
63  TX_ASSERT_MSG(contains(key), "Key does not exist.");
64  std::size_t idx_of_removed_element = _key_to_index_map[key];
65  std::size_t idx_of_last_element = _size - 1;
66 
67  if (idx_of_removed_element == idx_of_last_element) {
68  _values[idx_of_removed_element].~Value();
69  } else {
70  _values[idx_of_removed_element] = std::move(_values[idx_of_last_element]);
71 
72  Key key_of_last_element = _index_to_key_map[idx_of_last_element];
73  _key_to_index_map[key_of_last_element] = idx_of_removed_element;
74  _index_to_key_map[idx_of_removed_element] = key_of_last_element;
75  }
76 
77  _key_to_index_map.erase(key);
78  _index_to_key_map.erase(idx_of_last_element);
79 
80  --_size;
81  }
